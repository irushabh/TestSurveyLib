@using SurveyLib.Models
@namespace SurveyLib.Components

<div class="survey-runner">
    @if (_submitted)
    {
        <div class="thank-you">
            <div class="check-icon">✓</div>
            <h2>@(_isEditMode ? "Response updated!" : "Thank you!")</h2>
            <p>@(_isEditMode ? "Your changes have been saved." : "Your response has been recorded.")</p>
        </div>
    }
    else
    {
        <div class="runner-header">
            <h2>@SurveyTitle</h2>
            <div class="runner-header-meta">
                <p class="question-count">@Questions.Count question@(Questions.Count != 1 ? "s" : "")</p>
                @if (_isEditMode)
                {
                    <span class="edit-badge">✎ Editing response</span>
                }
            </div>
        </div>

        @if (_isEditMode)
        {
            <div class="edit-notice">
                <span>📋 Your previous answers have been loaded. Make any changes below.</span>
            </div>
        }

        <div class="questions-list questions-grid">
            @foreach (var (q, i) in Questions.Select((q, i) => (q, i)))
            {
                <div class="runner-question @(q.FullWidth ? "col-full" : "col-half")">
                    <p class="question-label">
                        <span class="q-num">@(i + 1).</span>
                        @q.Text
                        @if (q.IsRequired) { <span class="required">*</span> }
                    </p>

                    @switch (q.Type)
                    {
                        case QuestionType.Text:
                            <textarea rows="3"
                                      placeholder="Your answer..."
                                      value="@_responses[q.Id].TextAnswer"
                                      @onchange="e => SetText(q.Id, e)"></textarea>
                            break;

                        case QuestionType.MultipleChoice:
                            @foreach (var opt in q.Options)
                            {
                                <label class="radio-label">
                                    <input type="radio"
                                           name="@q.Id"
                                           value="@opt"
                                           checked="@(_responses[q.Id].SelectedOptions.Contains(opt))"
                                           @onchange="e => SetSingle(q.Id, e)" />
                                    @opt
                                </label>
                            }
                            break;

                        case QuestionType.Checkbox:
                            @foreach (var opt in q.Options)
                            {
                                <label class="checkbox-label">
                                    <input type="checkbox"
                                           checked="@(_responses[q.Id].SelectedOptions.Contains(opt))"
                                           @onchange="e => ToggleCheckbox(q.Id, opt, e)" />
                                    @opt
                                </label>
                            }
                            break;

                        case QuestionType.Rating:
                            <div class="star-rating">
                                @for (int s = 1; s <= q.MaxRating; s++)
                                {
                                    var star = s;
                                    <button class="star @(_responses[q.Id].RatingAnswer >= star ? "active" : "")"
                                            @onclick="() => SetRating(q.Id, star)">★</button>
                                }
                            </div>
                            break;

                        case QuestionType.YesNo:
                            <div class="yesno-group">
                                <button class="btn-yesno @(_responses[q.Id].YesNoAnswer == true ? "active" : "")"
                                        @onclick="() => SetYesNo(q.Id, true)">Yes</button>
                                <button class="btn-yesno @(_responses[q.Id].YesNoAnswer == false ? "active" : "")"
                                        @onclick="() => SetYesNo(q.Id, false)">No</button>
                            </div>
                            break;
                    }
                </div>
            }
        </div>

        @if (_validationError is not null)
        {
            <p class="validation-error">⚠ @_validationError</p>
        }

        <div class="runner-footer">
            @if (_isEditMode && _isDirty)
            {
                <button class="btn secondary" @onclick="DiscardChanges">↩ Discard Changes</button>
            }
            <button class="btn primary @(_isEditMode ? "" : "full-width")" @onclick="Submit">
                @(_isEditMode ? (_isDirty ? "💾 Save Changes" : "✓ No Changes") : "Submit Survey")
            </button>
        </div>
    }
</div>
